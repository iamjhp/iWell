import Model from '../model/mapModel.js';
import Map from '../view/mapView.js';

/**
 * @class controller
 *
 * Controller for the iWell Application.
 * Links the user input and the view output.
 */
class MapController {

    /**
     * Initializes a new instance of the MapController with the given model & view instances.
     * @param model model
     * @param view view
     */
    constructor(model, view) {
        this.model = model;
        this.view = view;
        this.view.find_closest_Well = this.#find_closest_Well.bind(this);
    }

    #find_closest_Well(event) {
        let closestId = this.model.getClosestWell(event);            // data from model
        this.view.showClosestWellInfo(closestId);                 // data to view
    }
}

const app = new MapController(new Model(), new Map());
let wells = app.model.getWells();
let userPosition = app.model.getUserLocation();
app.view.initMap(wells, userPosition);
