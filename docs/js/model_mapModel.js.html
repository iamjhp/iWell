async function getWellsFromDB() {

    /**
     * Calculates the closest well from the user's location and returns the id of this well.
     * @param event
     * @returns {number} id of the closest well
     */
    getClosestWell(event) {
        let distances = [];
        let closest = -1;
        let idx = -1;
        for (let i = 0; i < this.#well.length; i++) {
            let wellCoordinate = new google.maps.LatLng(this.#well[i].latitude, this.#well[i].longitude)
            let zhaw = new google.maps.LatLng(zhawCoordinates.lat, zhawCoordinates.lng)
            let d = google.maps.geometry.spherical.computeDistanceBetween(wellCoordinate, zhaw);
            distances[i] = d;
            if (idx == -1 || d < distances[idx]) {
                idx = i;
                closest = this.#well[i].id;
            }
        }
        return closest;
    }
}
